class node:
    def __init__(self,x):
        self.data=x
        self.next=None
class Linkedlist:
    def __init__(self):
        self.head=self.tail=None
    def create(self,x):
        n=node(x)
        if self.head==None:
            self.head=self.tail=n
        else:
            self.tail.next=n
            self.tail=n
    def show(self):
        temp=self.head
        while temp!=None:
            print(temp.data)
            temp=temp.next
l=Linkedlist()
while True:
    x=int(input())
    if x==-1:
        break
    else:
        l.create(x)
l.show()
